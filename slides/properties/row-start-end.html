grid-row-start

.item1 {
	grid-column-start: 1;
	grid-column-end: 3;
	grid-row-start: 1;
	grid-row-end: 4;
}

Shorthand: grid-row

	
.item1 {
	grid-column: 1 / 3;
	grid-row: 1 / 4;
}
